 package com.vangaurd.tradereporting.service;
 
 import com.vangaurd.tradereporting.exception.XmlParseException;
 import com.vangaurd.tradereporting.model.Event;
 import com.vangaurd.tradereporting.service.impl.XmlParserServiceImpl;
 import org.junit.jupiter.api.Test;
 import org.junit.jupiter.api.extension.ExtendWith;
 import org.mockito.InjectMocks;
 import org.mockito.junit.jupiter.MockitoExtension;
 
 import java.math.BigDecimal;
 import java.util.List;
 
 import static org.junit.jupiter.api.Assertions.*;
 
 @ExtendWith(MockitoExtension.class)
 public class XmlParserServiceTest {
     @InjectMocks
     private XmlParserServiceImpl xmlParserService;
 
     @Test
     void testParseValidXmlFile() throws XmlParseException {
         String validFilePath = "src/test/resources/";
         List<Event> events = xmlParserService.parseXmlFile(validFilePath);
         assertFalse(events.isEmpty());
         assertEquals(1, events.size());
         Event event = events.get(0);
         assertEquals("LEFT_BANK", event.getBuyerParty());
         assertEquals("EMU_BANK", event.getSellerParty());
         assertEquals(new BigDecimal("100.00"), event.getPremiumAmount());
         assertEquals("AUD", event.getPremiumCurrency());
     }
 
     @Test
     void testParseInvalidXmlFile() {
         String invalidFilePath = "src/test/resources/invalid_event.xml";
         assertThrows(XmlParseException.class, () -> xmlParserService.parseXmlFile(invalidFilePath));
     }
 }
