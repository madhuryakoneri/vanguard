 package com.vangaurd.tradereporting.service.impl;
 
 import com.vangaurd.tradereporting.exception.EventProcessingException;
 import com.vangaurd.tradereporting.exception.InvalidFilterCriteriaException;
 import com.vangaurd.tradereporting.exception.XmlParseException;
 import com.vangaurd.tradereporting.filters.IEventFilter;
 import com.vangaurd.tradereporting.model.Event;
 import com.vangaurd.tradereporting.repository.EventRepository;
 import com.vangaurd.tradereporting.service.IEventService;
 import com.vangaurd.tradereporting.service.IXmlParserService;
 import org.springframework.stereotype.Service;
 
 import java.util.List;
 
 @Service
 public class EventServiceImpl implements IEventService {
     private final EventRepository eventRepository;
     private final IXmlParserService xmlParserService;
     private final List<IEventFilter> filters;
 
     public EventServiceImpl(EventRepository eventRepository, IXmlParserService xmlParserService, List<IEventFilter> filters) {
         this.eventRepository = eventRepository;
         this.xmlParserService = xmlParserService;
         this.filters = filters;
     }
     @Override
     public void processXmlFile(String filePath) throws Exception {
         try {
             List<Event> events = xmlParserService.parseXmlFile(filePath);
             eventRepository.saveAll(events);
         } catch (XmlParseException e) {
             throw e;
         } catch (Exception e) {
             throw new EventProcessingException("Error processing events: " + e.getMessage());
         }
     }
 
     @Override
     public List<Event> getFilteredEvents() {
         try {
             List<Event> allEvents = eventRepository.findAll();
             return allEvents.stream()
                     .filter(event -> filters.stream().allMatch(filter -> filter.apply(event)))
                     .toList();
         } catch (Exception e) {
             throw new InvalidFilterCriteriaException("Error applying filters: " + e.getMessage());
         }
     }
 }
