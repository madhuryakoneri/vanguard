     @Override
     public List<Event> parseXmlFile(String directoryPath) throws XmlParseException {
         List<Event> events = new ArrayList<>();
         File directory = new File(directoryPath);
         if (directory.isDirectory()) {
             for (File file : directory.listFiles()) {
                 if (file.isFile() && file.getName().endsWith(".xml")) {
                     try {
                         events.add(parseSingleXmlFile(file.getAbsolutePath()));
                     } catch (XmlParseException e) {
                         throw new XmlParseException("Error parsing file: " + file.getName() + e);
                     }
                 }
             }
         } else {
             throw new XmlParseException("Invalid directory path: " + directoryPath);
         }
         return events;
     }
